109 void mkdir_p(const char* path) {
110  int error = 0;
111  printf("mkdir %s\n", path);
112  if (mkdir(path, 0777)) {
113  error = errno;
114  if (error == ENOENT) {
115  char *sep = strrchr((char*)path, '/');
116  if (sep == NULL) {
117  sep = strrchr((char*)path, SLASH[0]);
118  }
119  if (sep != NULL) {
120  *sep = '\0';
121  if (strlen(path) > 0) {
122  // will be null if the we reach the first /
123  mkdir_p(path);
124  }
125  *sep = '/';
126  mkdir(path, 0777);
127  }
128  } else if (error != EEXIST) {
129  if (mkdir(path, 0777)) {
130  printf("Error while trying to create '%s'\n", path);
131  }
132  }
133  }
134 }

139 void copy_dir(const char* source_folder, const char* target_folder)
140 {
141  // make the parentdir of the target folder
142  char* ptarget = _strdup(target_folder);
143  parentdir(ptarget);
144  mkdir_p(ptarget);
145  // convert source and target folder strings to windows wide strings
146  wchar_t *wsource_folder = new wchar_t[strlen(source_folder) + 1];
147  size_t convertedChars = 0;
148  mbstowcs_s(&convertedChars, wsource_folder, strlen(source_folder) + 1, source_folder, _TRUNCATE);
149  wchar_t *wtarget_folder = new wchar_t[strlen(target_folder) + 1];
150  mbstowcs_s(&convertedChars, wtarget_folder, strlen(target_folder) + 1, target_folder, _TRUNCATE);
151  WCHAR sf[MAX_PATH + 1];
152  WCHAR tf[MAX_PATH + 1];
153  wcscpy_s(sf, MAX_PATH, wsource_folder);
154  wcscpy_s(tf, MAX_PATH, wtarget_folder);
155  // ensure we have double-null terminated strings like Windows docs demand
156  sf[lstrlenW(sf) + 1] = 0;
157  tf[lstrlenW(tf) + 1] = 0;
158  // first delete the target_folder, if it exists
159  SHFILEOPSTRUCTW deleteS = { 0 };
160  deleteS.wFunc = FO_DELETE;
161  deleteS.pTo = tf;
162  deleteS.pFrom = tf;
163  deleteS.fFlags = FOF_SILENT | FOF_NOCONFIRMMKDIR | FOF_NOCONFIRMATION | FOF_NOERRORUI | FOF_NO_UI;
164  SHFileOperationW(&deleteS);
165  // now copy the new folder in its place
166  SHFILEOPSTRUCTW s = { 0 };
167  s.wFunc = FO_COPY;
168  s.pTo = tf;
169  s.pFrom = sf;
170  s.fFlags = FOF_SILENT | FOF_NOCONFIRMMKDIR | FOF_NOCONFIRMATION | FOF_NOERRORUI | FOF_NO_UI;
171  SHFileOperationW(&s);
172 }

174 int copy_file(const char* src_path, const struct stat* sb, int typeflag) {
175  char dst_path[BUFF_SIZE];
176  printf("%s to %s\n", src_path, dst_root);
177  strcpy(dst_path, dst_root);
178  strcat(dst_path, src_path + strlen(src_root));
179  switch(typeflag) {
180  case FTW_D:
181  mkdir_p(dst_path);
182  break;
183  case FTW_F:
184  mkdir_p(parentdir(strdup(dst_path)));
185  FILE* in = fopen(src_path, "rb");
186  FILE* out = fopen(dst_path, "wb");
187  char buf[4096];
188  int c = 0;
189  if (!in) {
190  error("Extraction error", "Could not open source folder for reading.");
191  }
192  if (!out) {
193  error("Extraction error", "Could not open data folder for writing.");
194  }
195  while ((c = fread(buf, sizeof(char), 4096, in))) {
196  fwrite(buf, sizeof(char), c, out);
197  }
198  fclose(in);
199  fclose(out);
200  break;
201  }
202  return 0;
203 }
