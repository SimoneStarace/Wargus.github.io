505 local function handle_breakpoint(peer)
506  -- check if the buffer has the beginning of SETB/DELB command;
507  -- this is to avoid reading the entire line for commands that
508  -- don't need to be handled here.
509  if not buf or not (buf:sub(1,1) == 'S' or buf:sub(1,1) == 'D') then return end
510 
511  -- check second character to avoid reading STEP or other S* and D* commands
512  if #buf == 1 then buf = buf .. readnext(peer, 1) end
513  if buf:sub(2,2) ~= 'E' then return end
514 
515  -- need to read few more characters
516  buf = buf .. readnext(peer, 5-#buf)
517  if buf ~= 'SETB ' and buf ~= 'DELB ' then return end
518 
519  local res, _, partial = peer:receive("*l") -- get the rest of the line; blocking
520  if not res then
521  if partial then buf = buf .. partial end
522  return
523  end
524 
525  local _, _, cmd, file, line = (buf..res):find("^([A-Z]+)%s+(.-)%s+(%d+)%s*$")
526  if cmd == 'SETB' then set_breakpoint(file, tonumber(line))
527  elseif cmd == 'DELB' then remove_breakpoint(file, tonumber(line))
528  else
529  -- this looks like a breakpoint command, but something went wrong;
530  -- return here to let the "normal" processing to handle,
531  -- although this is likely to not go well.
532  return
533  end
534 
535  buf = nil
536 end

538 local function normalize_path(file)
539  local n
540  repeat
541  file, n = file:gsub("/+%.?/+","/") -- remove all `//` and `/./` references
542  until n == 0
543  -- collapse all up-dir references: this will clobber UNC prefix (\\?\)
544  -- and disk on Windows when there are too many up-dir references: `D:\foo\..\..\bar`;
545  -- handle the case of multiple up-dir references: `foo/bar/baz/../../../more`;
546  -- only remove one at a time as otherwise `../../` could be removed;
547  repeat
548  file, n = file:gsub("[^/]+/%.%./", "", 1)
549  until n == 0
550  -- there may still be a leading up-dir reference left (as `/../` or `../`); remove it
551  return (file:gsub("^(/?)%.%./", "%1"))
552 end

554 local function debug_hook(event, line)
555  -- (1) LuaJIT needs special treatment. Because debug_hook is set for
556  -- *all* coroutines, and not just the one being debugged as in regular Lua
557  -- (http://lua-users.org/lists/lua-l/2011-06/msg00513.html),
558  -- need to avoid debugging mobdebug's own code as LuaJIT doesn't
559  -- always correctly generate call/return hook events (there are more
560  -- calls than returns, which breaks stack depth calculation and
561  -- 'step' and 'step over' commands stop working; possibly because
562  -- 'tail return' events are not generated by LuaJIT).
563  -- the next line checks if the debugger is run under LuaJIT and if
564  -- one of debugger methods is present in the stack, it simply returns.
565  -- ngx_lua/Openresty requires a slightly different handling, as it
566  -- creates a coroutine wrapper, so this processing needs to be skipped.
567  if jit and not (ngx and type(ngx) == "table" and ngx.say) then
568  -- when luajit is compiled with LUAJIT_ENABLE_LUA52COMPAT,
569  -- coroutine.running() returns non-nil for the main thread.
570  local coro, main = coroutine.running()
571  if not coro or main then coro = 'main' end
572  local disabled = coroutines[coro] == false
573  or coroutines[coro] == nil and coro ~= (coro_debugee or 'main')
574  if coro_debugee and disabled or not coro_debugee and (disabled or in_debugger()) then
575  return
576  end
577  end
578 
579  -- (2) check if abort has been requested and it's safe to abort
580  if abort and is_safe(stack_level) then error(abort) end
581 
582  -- (3) also check if this debug hook has not been visited for any reason.
583  -- this check is needed to avoid stepping in too early
584  -- (for example, when coroutine.resume() is executed inside start()).
585  if not seen_hook and in_debugger() then return end
586 
587  if event == "call" then
588  stack_level = stack_level + 1
589  elseif event == "return" or event == "tail return" then
590  stack_level = stack_level - 1
591  elseif event == "line" then
592  if mobdebug.linemap then
593  local ok, mappedline = pcall(mobdebug.linemap, line, debug.getinfo(2, "S").source)
594  if ok then line = mappedline end
595  if not line then return end
596  end
597 
598  -- may need to fall through because of the following:
599  -- (1) step_into
600  -- (2) step_over and stack_level <= step_level (need stack_level)
601  -- (3) breakpoint; check for line first as it's known; then for file
602  -- (4) socket call (only do every Xth check)
603  -- (5) at least one watch is registered
604  if not (
605  step_into or step_over or breakpoints[line] or watchescnt > 0
606  or is_pending(server)
607  ) then checkcount = checkcount + 1; return end
608 
609  checkcount = mobdebug.checkcount -- force check on the next command
610 
611  -- this is needed to check if the stack got shorter or longer.
612  -- unfortunately counting call/return calls is not reliable.
613  -- the discrepancy may happen when "pcall(load, '')" call is made
614  -- or when "error()" is called in a function.
615  -- in either case there are more "call" than "return" events reported.
616  -- this validation is done for every "line" event, but should be "cheap"
617  -- as it checks for the stack to get shorter (or longer by one call).
618  -- start from one level higher just in case we need to grow the stack.
619  -- this may happen after coroutine.resume call to a function that doesn't
620  -- have any other instructions to execute. it triggers three returns:
621  -- "return, tail return, return", which needs to be accounted for.
622  stack_level = stack_depth(stack_level+1)
623 
624  local caller = debug.getinfo(2, "S")
625 
626  -- grab the filename and fix it if needed
627  local file = lastfile
628  if (lastsource ~= caller.source) then
629  file, lastsource = caller.source, caller.source
630  -- technically, users can supply names that may not use '@',
631  -- for example when they call loadstring('...', 'filename.lua').
632  -- Unfortunately, there is no reliable/quick way to figure out
633  -- what is the filename and what is the source code.
634  -- If the name doesn't start with `@`, assume it's a file name if it's all on one line.
635  if find(file, "^@") or not find(file, "[\r\n]") then
636  file = gsub(gsub(file, "^@", ""), "\\", "/")
637  -- normalize paths that may include up-dir or same-dir references
638  -- if the path starts from the up-dir or reference,
639  -- prepend `basedir` to generate absolute path to keep breakpoints working.
640  -- ignore qualified relative path (`D:../`) and UNC paths (`\\?\`)
641  if find(file, "^%.%./") then file = basedir..file end
642  if find(file, "/%.%.?/") then file = normalize_path(file) end
643  -- need this conversion to be applied to relative and absolute
644  -- file names as you may write "require 'Foo'" to
645  -- load "foo.lua" (on a case insensitive file system) and breakpoints
646  -- set on foo.lua will not work if not converted to the same case.
647  if iscasepreserving then file = string.lower(file) end
648  if find(file, "^%./") then file = sub(file, 3) end
649  -- remove basedir, so that breakpoints are checked properly
650  file = gsub(file, "^"..q(basedir), "")
651  -- some file systems allow newlines in file names; remove these.
652  file = gsub(file, "\n", ' ')
653  else
654  file = mobdebug.line(file)
655  end
656 
657  -- set to true if we got here; this only needs to be done once per
658  -- session, so do it here to at least avoid setting it for every line.
659  seen_hook = true
660  lastfile = file
661  end
662 
663  if is_pending(server) then handle_breakpoint(server) end
664 
665  local vars, status, res
666  if (watchescnt > 0) then
667  vars = capture_vars(1)
668  for index, value in pairs(watches) do
669  setfenv(value, vars)
670  local ok, fired = pcall(value)
671  if ok and fired then
672  status, res = cororesume(coro_debugger, events.WATCH, vars, file, line, index)
673  break -- any one watch is enough; don't check multiple times
674  end
675  end
676  end
677 
678  -- need to get into the "regular" debug handler, but only if there was
679  -- no watch that was fired. If there was a watch, handle its result.
680  local getin = (status == nil) and
681  (step_into
682  -- when coroutine.running() return `nil` (main thread in Lua 5.1),
683  -- step_over will equal 'main', so need to check for that explicitly.
684  or (step_over and step_over == (coroutine.running() or 'main') and stack_level <= step_level)
685  or has_breakpoint(file, line)
686  or is_pending(server))
687 
688  if getin then
689  vars = vars or capture_vars(1)
690  step_into = false
691  step_over = false
692  status, res = cororesume(coro_debugger, events.BREAK, vars, file, line)
693  end
694 
695  -- handle 'stack' command that provides stack() information to the debugger
696  while status and res == 'stack' do
697  -- resume with the stack trace and variables
698  if vars then restore_vars(vars) end -- restore vars so they are reflected in stack values
699  status, res = cororesume(coro_debugger, events.STACK, stack(3), file, line)
700  end
701 
702  -- need to recheck once more as resume after 'stack' command may
703  -- return something else (for example, 'exit'), which needs to be handled
704  if status and res and res ~= 'stack' then
705  if not abort and res == "exit" then mobdebug.onexit(1, true); return end
706  if not abort and res == "done" then mobdebug.done(); return end
707  abort = res
708  -- only abort if safe; if not, there is another (earlier) check inside
709  -- debug_hook, which will abort execution at the first safe opportunity
710  if is_safe(stack_level) then error(abort) end
711  elseif not status and res then
712  error(res, 2) -- report any other (internal) errors back to the application
713  end
714 
715  if vars then restore_vars(vars) end
716 
717  -- last command requested Step Over/Out; store the current thread
718  if step_over == true then step_over = coroutine.running() or 'main' end
719  end
720 end

722 local function stringify_results(params, status, ...)
723  if not status then return status, ... end -- on error report as it
724 
725  params = params or {}
726  if params.nocode == nil then params.nocode = true end
727  if params.comment == nil then params.comment = 1 end
728 
729  local t = {...}
730  for i,v in pairs(t) do -- stringify each of the returned values
731  local ok, res = pcall(mobdebug.line, v, params)
732  t[i] = ok and res or ("%q"):format(res):gsub("\010","n"):gsub("\026","\\026")
733  end
734  -- stringify table with all returned values
735  -- this is done to allow each returned value to be used (serialized or not)
736  -- intependently and to preserve "original" comments
737  return pcall(mobdebug.dump, t, {sparse = false})
738 end

765 local function debugger_loop(sev, svars, sfile, sline)
766  local command
767  local eval_env = svars or {}
768  local function emptyWatch () return false end
769  local loaded = {}
770  for k in pairs(package.loaded) do loaded[k] = true end
771 
772  while true do
773  local line, err
774  if mobdebug.yield and server.settimeout then server:settimeout(mobdebug.yieldtimeout) end
775  while true do
776  line, err = server:receive("*l")
777  if not line then
778  if err == "timeout" then
779  if mobdebug.yield then mobdebug.yield() end
780  elseif err == "closed" then
781  error("Debugger connection closed", 0)
782  else
783  error(("Unexpected socket error: %s"):format(err), 0)
784  end
785  else
786  -- if there is something in the pending buffer, prepend it to the line
787  if buf then line = buf .. line; buf = nil end
788  break
789  end
790  end
791  if server.settimeout then server:settimeout() end -- back to blocking
792  command = string.sub(line, string.find(line, "^[A-Z]+"))
793  if command == "SETB" then
794  local _, _, _, file, line = string.find(line, "^([A-Z]+)%s+(.-)%s+(%d+)%s*$")
795  if file and line then
796  set_breakpoint(file, tonumber(line))
797  server:send("200 OK\n")
798  else
799  server:send("400 Bad Request\n")
800  end
801  elseif command == "DELB" then
802  local _, _, _, file, line = string.find(line, "^([A-Z]+)%s+(.-)%s+(%d+)%s*$")
803  if file and line then
804  remove_breakpoint(file, tonumber(line))
805  server:send("200 OK\n")
806  else
807  server:send("400 Bad Request\n")
808  end
809  elseif command == "EXEC" then
810  -- extract any optional parameters
811  local params = string.match(line, "--%s*(%b{})%s*$")
812  local _, _, chunk = string.find(line, "^[A-Z]+%s+(.+)$")
813  if chunk then
814  local func, res = mobdebug.loadstring(chunk)
815  local status
816  if func then
817  local pfunc = params and loadstring("return "..params) -- use internal function
818  params = pfunc and pfunc()
819  params = (type(params) == "table" and params or {})
820  local stack = tonumber(params.stack)
821  -- if the requested stack frame is not the current one, then use a new capture
822  -- with a specific stack frame: `capture_vars(0, coro_debugee)`
823  local env = stack and coro_debugee and capture_vars(stack-1, coro_debugee) or eval_env
824  setfenv(func, env)
825  status, res = stringify_results(params, pcall(func, unpack(rawget(env,'...') or {})))
826  end
827  if status then
828  if mobdebug.onscratch then mobdebug.onscratch(res) end
829  server:send("200 OK " .. tostring(#res) .. "\n")
830  server:send(res)
831  else
832  -- fix error if not set (for example, when loadstring is not present)
833  if not res then res = "Unknown error" end
834  server:send("401 Error in Expression " .. tostring(#res) .. "\n")
835  server:send(res)
836  end
837  else
838  server:send("400 Bad Request\n")
839  end
840  elseif command == "LOAD" then
841  local _, _, size, name = string.find(line, "^[A-Z]+%s+(%d+)%s+(%S.-)%s*$")
842  size = tonumber(size)
843 
844  if abort == nil then -- no LOAD/RELOAD allowed inside start()
845  if size > 0 then server:receive(size) end
846  if sfile and sline then
847  server:send("201 Started " .. sfile .. " " .. tostring(sline) .. "\n")
848  else
849  server:send("200 OK 0\n")
850  end
851  else
852  -- reset environment to allow required modules to load again
853  -- remove those packages that weren't loaded when debugger started
854  for k in pairs(package.loaded) do
855  if not loaded[k] then package.loaded[k] = nil end
856  end
857 
858  if size == 0 and name == '-' then -- RELOAD the current script being debugged
859  server:send("200 OK 0\n")
860  coroyield("load")
861  else
862  -- receiving 0 bytes blocks (at least in luasocket 2.0.2), so skip reading
863  local chunk = size == 0 and "" or server:receive(size)
864  if chunk then -- LOAD a new script for debugging
865  local func, res = mobdebug.loadstring(chunk, "@"..name)
866  if func then
867  server:send("200 OK 0\n")
868  debugee = func
869  coroyield("load")
870  else
871  server:send("401 Error in Expression " .. tostring(#res) .. "\n")
872  server:send(res)
873  end
874  else
875  server:send("400 Bad Request\n")
876  end
877  end
878  end
879  elseif command == "SETW" then
880  local _, _, exp = string.find(line, "^[A-Z]+%s+(.+)%s*$")
881  if exp then
882  local func, res = mobdebug.loadstring("return(" .. exp .. ")")
883  if func then
884  watchescnt = watchescnt + 1
885  local newidx = #watches + 1
886  watches[newidx] = func
887  server:send("200 OK " .. tostring(newidx) .. "\n")
888  else
889  server:send("401 Error in Expression " .. tostring(#res) .. "\n")
890  server:send(res)
891  end
892  else
893  server:send("400 Bad Request\n")
894  end
895  elseif command == "DELW" then
896  local _, _, index = string.find(line, "^[A-Z]+%s+(%d+)%s*$")
897  index = tonumber(index)
898  if index > 0 and index <= #watches then
899  watchescnt = watchescnt - (watches[index] ~= emptyWatch and 1 or 0)
900  watches[index] = emptyWatch
901  server:send("200 OK\n")
902  else
903  server:send("400 Bad Request\n")
904  end
905  elseif command == "RUN" then
906  server:send("200 OK\n")
907 
908  local ev, vars, file, line, idx_watch = coroyield()
909  eval_env = vars
910  if ev == events.BREAK then
911  server:send("202 Paused " .. file .. " " .. tostring(line) .. "\n")
912  elseif ev == events.WATCH then
913  server:send("203 Paused " .. file .. " " .. tostring(line) .. " " .. tostring(idx_watch) .. "\n")
914  elseif ev == events.RESTART then
915  -- nothing to do
916  else
917  server:send("401 Error in Execution " .. tostring(#file) .. "\n")
918  server:send(file)
919  end
920  elseif command == "STEP" then
921  server:send("200 OK\n")
922  step_into = true
923 
924  local ev, vars, file, line, idx_watch = coroyield()
925  eval_env = vars
926  if ev == events.BREAK then
927  server:send("202 Paused " .. file .. " " .. tostring(line) .. "\n")
928  elseif ev == events.WATCH then
929  server:send("203 Paused " .. file .. " " .. tostring(line) .. " " .. tostring(idx_watch) .. "\n")
930  elseif ev == events.RESTART then
931  -- nothing to do
932  else
933  server:send("401 Error in Execution " .. tostring(#file) .. "\n")
934  server:send(file)
935  end
936  elseif command == "OVER" or command == "OUT" then
937  server:send("200 OK\n")
938  step_over = true
939 
940  -- OVER and OUT are very similar except for
941  -- the stack level value at which to stop
942  if command == "OUT" then step_level = stack_level - 1
943  else step_level = stack_level end
944 
945  local ev, vars, file, line, idx_watch = coroyield()
946  eval_env = vars
947  if ev == events.BREAK then
948  server:send("202 Paused " .. file .. " " .. tostring(line) .. "\n")
949  elseif ev == events.WATCH then
950  server:send("203 Paused " .. file .. " " .. tostring(line) .. " " .. tostring(idx_watch) .. "\n")
951  elseif ev == events.RESTART then
952  -- nothing to do
953  else
954  server:send("401 Error in Execution " .. tostring(#file) .. "\n")
955  server:send(file)
956  end
957  elseif command == "BASEDIR" then
958  local _, _, dir = string.find(line, "^[A-Z]+%s+(.+)%s*$")
959  if dir then
960  basedir = iscasepreserving and string.lower(dir) or dir
961  -- reset cached source as it may change with basedir
962  lastsource = nil
963  server:send("200 OK\n")
964  else
965  server:send("400 Bad Request\n")
966  end
967  elseif command == "SUSPEND" then
968  -- do nothing; it already fulfilled its role
969  elseif command == "DONE" then
970  coroyield("done")
971  return -- done with all the debugging
972  elseif command == "STACK" then
973  -- first check if we can execute the stack command
974  -- as it requires yielding back to debug_hook it cannot be executed
975  -- if we have not seen the hook yet as happens after start().
976  -- in this case we simply return an empty result
977  local vars, ev = {}
978  if seen_hook then
979  ev, vars = coroyield("stack")
980  end
981  if ev and ev ~= events.STACK then
982  server:send("401 Error in Execution " .. tostring(#vars) .. "\n")
983  server:send(vars)
984  else
985  local params = string.match(line, "--%s*(%b{})%s*$")
986  local pfunc = params and loadstring("return "..params) -- use internal function
987  params = pfunc and pfunc()
988  params = (type(params) == "table" and params or {})
989  if params.nocode == nil then params.nocode = true end
990  if params.sparse == nil then params.sparse = false end
991  -- take into account additional levels for the stack frames and data management
992  if tonumber(params.maxlevel) then params.maxlevel = tonumber(params.maxlevel)+4 end
993 
994  local ok, res = pcall(mobdebug.dump, vars, params)
995  if ok then
996  server:send("200 OK " .. tostring(res) .. "\n")
997  else
998  server:send("401 Error in Execution " .. tostring(#res) .. "\n")
999  server:send(res)
1000  end
1001  end
1002  elseif command == "OUTPUT" then
1003  local _, _, stream, mode = string.find(line, "^[A-Z]+%s+(%w+)%s+([dcr])%s*$")
1004  if stream and mode and stream == "stdout" then
1005  -- assign "print" in the global environment
1006  local default = mode == 'd'
1007  genv.print = default and iobase.print or corowrap(function()
1008  -- wrapping into coroutine.wrap protects this function from
1009  -- being stepped through in the debugger.
1010  -- don't use vararg (...) as it adds a reference for its values,
1011  -- which may affect how they are garbage collected
1012  while true do
1013  local tbl = {coroutine.yield()}
1014  if mode == 'c' then iobase.print(unpack(tbl)) end
1015  for n = 1, #tbl do
1016  tbl[n] = select(2, pcall(mobdebug.line, tbl[n], {nocode = true, comment = false})) end
1017  local file = table.concat(tbl, "\t").."\n"
1018  server:send("204 Output " .. stream .. " " .. tostring(#file) .. "\n" .. file)
1019  end
1020  end)
1021  if not default then genv.print() end -- "fake" print to start printing loop
1022  server:send("200 OK\n")
1023  else
1024  server:send("400 Bad Request\n")
1025  end
1026  elseif command == "EXIT" then
1027  server:send("200 OK\n")
1028  coroyield("exit")
1029  else
1030  server:send("400 Bad Request\n")
1031  end
1032  end
1033 end
